  html`
    <sgds-alert variant="success" alertClasses="d-flex align-items-center">
      <sl-icon

        class="flex-shrink-0 me-4 align-self-start mt-2"
      ></sl-icon>
      <div>

        <p>

export const TemplateWithButton = (args) =>
  html`
    <sgds-alert
      id="alertExample"
      variant="warning"
      alertClasses="d-flex align-items-center"
      dismissible
      ><sl-icon name="info-circle" class="flex-shrink-0 me-4"></sl-icon>
      <div>
        This is an Alert component. You may add the dismissible option.
      </div></sgds-alert
    >
    <br />
    <sgds-button variant="danger" refId="alertExample" methodType="toggleShow"
      >Show Alert</sgds-button
    >
  `;

## Additional Content

Alerts can contain whatever content you like. Headers, paragraphs, dividers etc.

<Canvas>
  <Story name='Additional Content'>{TemplateAddContent.bind({})}</Story>
</Canvas>

## Dismissible

Add the dismissible prop to add a dismiss button to the Alert. An `sgds-button` with id reference `refId` prop to the alert can also be used to show the dismissed alert with `methodType` prop of value `toggleShow`.

<Canvas>
  <Story
    name='Dismissible'
    args={{ alertClasses: "d-flex align-items-center" }}
  >
    {TemplateWithButton.bind({})}
  </Story>
</Canvas>
